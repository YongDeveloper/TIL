# 알고리즘
[백준 12100번:2048(Easy)](https://www.acmicpc.net/problem/12100)
```java
package simulation_;

import java.io.*;
import java.util.*;

public class Main_12100_Easy2048 {

	private static int[][] map;
	private static int N, max = Integer.MIN_VALUE;

	public static void main(String[] args) throws IOException {

		BufferedReader br = new BufferedReader(new InputStreamReader(System.in));
		N = Integer.parseInt(br.readLine());
		map = new int [N][N];
		
		for (int i = 0; i < N; i++) {
			StringTokenizer st = new StringTokenizer(br.readLine());
			for (int j = 0; j < N; j++) {
				map[i][j] = Integer.parseInt(st.nextToken());
				if(max < map[i][j]) max = map[i][j];
			}
		}
		DFS(map, 0);
		System.out.println(max);
	}//main
	
	public static void DFS(int [][] map, int cnt) {
		if(cnt==5) return;
		DFS(Up(map), cnt+1);
		DFS(Down(map), cnt+1);
		DFS(Left(map), cnt+1);
		DFS(Right(map), cnt+1);
	}//DFS
	
	public static int[][] Up (int [][] map) {
		int [][] result = new int [N][N];
		Stack <Integer> stack = new Stack<>();
		boolean check = false;
		
		for (int j = 0; j < N; j++) {
			for (int i = 0; i < N; i++) {
				if(map[i][j] != 0) {
					if(stack.isEmpty()) stack.push(map[i][j]);
					else {
						if(!check && stack.peek() == map[i][j]) {
							stack.push(stack.pop() + map[i][j]);
							check = true;
							if(max < stack.peek()) max = stack.peek();
							continue;
						}else stack.push(map[i][j]);
					}
					check = false;
				}
			}
			while(!stack.isEmpty()) {
				int push = stack.size() -1;
				result[push][j] = stack.pop();
			}
		}
		return result;
	}//Up
	
	public static int[][] Down (int [][] map) {
		int [][] result = new int [N][N];
		Stack <Integer> stack = new Stack<>();
		boolean check = false;
		
		for (int j = 0; j < N; j++) {
			for (int i = N-1; i >=0 ; i--) {
				if(map[i][j] != 0) {
					if(stack.isEmpty()) stack.push(map[i][j]);
					else {
						if(!check && stack.peek()==map[i][j]) {
							stack.push(stack.pop() + map[i][j]);
							check = true;
							if(max < stack.peek()) max = stack.peek();
							continue;
						}else stack.push(map[i][j]);
					}
					check = false;
				}
			}
			while(!stack.isEmpty()) {
				int push = N - stack.size();
				result[push][j] = stack.pop();
			}
		}
		return result;
	}//Down
		
	public static int[][] Left (int [][] map) {
		int [][] result = new int [N][N];
		Stack <Integer> stack = new Stack<>();
		boolean check = false;
		
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				if(map[i][j]!=0) {
					if(stack.isEmpty()) stack.push(map[i][j]);
					else {
						if(!check && stack.peek() == map[i][j]) {
							stack.push(stack.pop() + map[i][j]);
							check = true;
							if(max < stack.peek()) max = stack.peek();
							continue;
						}else stack.push(map[i][j]);
					}
					check = false;
				}
			}
			while(!stack.isEmpty()) {
				int push = stack.size() -1;
				result[i][push] = stack.pop();
			}
		}
		return result;
	}//Left
	
	public static int[][] Right (int [][] map) {
		int [][] result = new int [N][N];
		Stack <Integer> stack = new Stack<>();
		boolean check = false;
		
		for (int i = 0; i < N; i++) {
			for (int j = N-1; j >= 0; j--) {
				if(map[i][j]!=0) {
					if(stack.isEmpty()) stack.push(map[i][j]);
					else {
						if(!check && stack.peek() == map[i][j]) {
							stack.push(stack.pop() + map[i][j]);
							check = true;
							if(max < stack.peek()) max = stack.peek();
							continue;
						}else stack.push(map[i][j]);
					}
					check = false;
				}
			}
			while(!stack.isEmpty()) {
				int push = N - stack.size();
				result[i][push] = stack.pop();
			}
		}
		return result;
	}//Right
	public static void Print(int [][] map) {
		for (int i = 0; i < N; i++) {
			for (int j = 0; j < N; j++) {
				System.out.print(map[i][j]+" ");
			}
			System.out.println();
		}
		System.out.println();
	}
}//class
```
